function SendMessage(hub, channelId, message) {
  console.log("channelId: ", channelId, "send message: ", message);

  if (hub) {
    hub.invoke("SendMessageAsync", {
      ReceiverId: channelId,
      Content: message,
      IsChannel: true,
    });
  } else {
    console.error("Hub is not connected");
  }
}
